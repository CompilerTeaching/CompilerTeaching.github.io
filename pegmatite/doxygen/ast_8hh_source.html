318  virtual void construct(const InputRange &r, ASTStack &st)
319  {
320  if (st.empty() && OPT)
321  {
322  return;
323  }
324  assert(!st.empty() && "Stack must not be empty");
325  ASTStackEntry &e = st.back();
326  const InputRange &childRange = e.first;
327  // If the entry isn't within the range of this, then it's just
328  // something of the same type that happens to be adjacent to this
329  // entry.
330  if ((childRange.begin() < r.begin()) ||
331  (childRange.end() > r.end()))
332  {
333  assert(OPT && "Required object not found");
334  return;
335  }
336  //get the node
337  ASTNode *node = e.second.get();
338 
339  //get the object
340  T *obj = node->get_as<T>();
341 
342  assert((obj || OPT) && "Required objects must exist!");
343  //if the object is optional, simply return
344  if (OPT && !obj)
345  {
346  return;
347  }
348  debug_log("Popped", st.size()-1, obj);
349  //set the new object
350  ptr.reset(obj);
351  //pop the node from the stack
352  st.back().second.release();
353  st.pop_back();
354  ptr->parent_node = container_node;
355  }
356 
357 private:
361  std::unique_ptr<T> ptr;
362 };

405  virtual void construct(const InputRange &r, ASTStack &st)
406  {
407  for(;;)
408  {
409  // If the stack is empty, don't fetch anything from it
410  if (st.empty()) break;
411  // Get the top entry on the stack
412  ASTStackEntry &e = st.back();
413  const InputRange &childRange = e.first;
414  // If the entry isn't within the range of this, then it's just
415  // something of the same type that happens to be adjacent to this
416  // entry.
417  if ((childRange.begin() < r.begin()) ||
418  (childRange.end() > r.end()))
419  {
420  break;
421  }
422 
423  //get the node
424  ASTNode *node = e.second.get();
425 
426  //get the object
427  T *obj = node->get_as<T>();
428 
429  //if the object was not not of the appropriate type,
430  //end the list parsing
431  if (!obj) return;
432  debug_log("Popped", st.size()-1, obj);
433 
434  //remove the node from the stack
435  e.second.release();
436  st.pop_back();
437 
438  //insert the object in the list, in reverse order
439  child_objects.push_front(std::unique_ptr<T>(obj));
440 
441  //set the object's parent
442  obj->parent_node = ASTMember::container();
443  }
444  }
