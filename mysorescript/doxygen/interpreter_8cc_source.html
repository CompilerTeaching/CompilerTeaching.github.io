315 Obj *Context::lookupSymbol(const std::string &name)
316 {
317  // If there's a symbol table stack, then look in the top one. We don't need
318  // to go deeper, because any bound variables should be added automatically
319  // to the top.
320  if (!symbols.empty())
321  {
322  auto top = symbols.back();
323  auto I = top->find(name);
324  if (I != top->end())
325  {
326  return I->second;
327  }
328  }
329  // If it's not local, then look in the global symbol table.
330  auto I = globalSymbols.find(name);
331  if (I != globalSymbols.end())
332  {
333  return I->second;
334  }
335  // It's not found, so return null.
336  return nullptr;
337 }

410 Obj Call::evaluateExpr(Interpreter::Context &c)
411 {
412  // Array of arguments.
413  Obj args[10];
414  // Get the callee, which is either a closure or some other object that will
415  // have a method on it invoked.
416  Obj obj = callee->evaluate(c);
417  assert(obj);
418  auto &argsAST = arguments->arguments;
419  size_t i=0;
420  // Evaluate each argument, in order, and pop them in the array.
421  for (auto &Arg : argsAST)
422  {
423  assert(i<(sizeof(args)/sizeof(Obj)));
424  args[i++] = Arg->evaluate(c);
425  }
426  // Get the class

428  // If there's no method, then we're trying to invoke a closure.
429  if (!method)

437  assert(sel);

451 void ClosureDecl::check()
452 {
453  // Don't do this if we've already done it.
454  if (checked)
455  {
456  return;
457  }
458  // Recursively collect all of the variables that are declared and referenced
459  // by statements in this closure.
460  body->collectVarUses(decls, boundVars);
461  // Parameters are not bound variables, they're explicitly passed in.
462  for (auto &param : parameters->arguments.objects())
463  {
464  boundVars.erase(param->name);
465  }
466  // Variables that are declared in this function are also not bound
467  // variables.
468  for (auto &decl : decls)
469  {
470  boundVars.erase(decl);
471  }
472  checked = true;
473 }

486 Obj ClosureDecl::evaluateExpr(Interpreter::Context &c)
487 {
488  check();
489  // Allocate the closure, with enough space for all of the bound variables
490  // at the end.
491  // Note that, in the current implementation, closures are always allocated
492  // on the heap. In many cases, closures do not persist longer than their
493  // wrapping function. In these cases, we could potentially allocate them
494  // on the stack, but making this determination requires a write barrier for
495  // the store.
496  Closure *C = gcAlloc<Closure>(boundVars.size() * sizeof(Obj));
497  // Set up the class pointer
498  C->isa = &ClosureClass;
499  // Set up the parameter count
500  size_t params = parameters->arguments.objects().size();
501  C->parameters = createSmallInteger(params);
502  C->AST = this;
503  C->invoke = compiledClosure ? compiledClosure :
504  Interpreter::closureTrampolines[params];
505  c.setSymbol(name->name, (Obj)C);
506  int i=0;
507  // Copy bound variables into the closure.
508  for (auto &var : boundVars)
509  {
510  C->boundVars[i++] = *c.lookupSymbol(var);
511  }
512  return (Obj)C;
513 }

515 Obj ClosureDecl::interpretMethod(Interpreter::Context &c, Method *mth, Obj self,
516  Selector sel, Obj *args)
517 {
518  check();
519  Class *cls = isInteger(self) ? &SmallIntClass : self->isa;
520  executionCount++;
521  // If we've interpreted this method enough times then try to compile it.
522  if (executionCount == compileThreshold)
523  {
524  mth->function = compileMethod(cls, c.globalSymbols);
525  compiledClosure = (ClosureInvoke)mth->function;
526  }
527  // If we now have a compiled version, try to execute it.
528  if (compiledClosure)
529  {
530  return callCompiledMethod((CompiledMethod)compiledClosure, self, sel,
531  args, parameters->arguments.objects().size());
532  }
533  // Create a new symbol table for this method.
534  Interpreter::SymbolTable closureSymbols;
535  c.pushSymbols(closureSymbols);
536  // Point the symbols for the arguments at our arguments array
537  size_t i=0;
538  for (auto &param : parameters->arguments)
539  {
540  c.setSymbol(param->name, &args[i++]);
541  }
542  Obj cmdObj = createSmallInteger(sel);
543  // Add self and cmd (receiver and selector) to the symbol table
544  c.setSymbol("self", &self);
545  c.setSymbol("cmd", &cmdObj);
546  // Add the addresses of the ivars in the self object to the symbol table.
547  Obj *ivars = ((Obj*)(&self->isa)) + 1;
548  for (int32_t i=0 ; i<cls->indexedIVarCount ; i++)
549  {
550  c.setSymbol(cls->indexedIVarNames[i], &ivars[i]);
551  }
552  // Interpret the statements in this method;
553  body->interpret(c);
554  // Return the return value. Make sure it's set back to nullptr after we've
555  // copied it so that we always return null from any method that doesn't
556  // explicitly return.
557  Obj retVal = c.retVal;
558  c.retVal = nullptr;
559  c.isReturning = false;
560  // Pop the symbols off the symbol table (very important, as they reference
561  // our stack frame!)
562  c.popSymbols();
563  return retVal;
564 }
565 Obj ClosureDecl::interpretClosure(Interpreter::Context &c, Closure *self,
566  Obj *args)
567 {
568  executionCount++;
569  // If we've interpreted this enough times, compile it.
570  if (executionCount == compileThreshold)
571  {
572  // Note that we don't pass any symbols other than the globals into the
573  // compiler, because all of the bound variables are already copied into
574  // the closure object when it is created.
575  self->invoke = compileClosure(c.globalSymbols);
576  compiledClosure = self->invoke;
577  }
578  // If we now have a compiled version, call it
579  if (compiledClosure)
580  {
581  callCompiledClosure(compiledClosure, self, args,
582  parameters->arguments.objects().size());
583  }
584  // Create a new symbol table for this closure
585  Interpreter::SymbolTable closureSymbols;
586  c.pushSymbols(closureSymbols);
587  size_t i=0;
588  // Add the parameters and the bound variables to the symbol table.
589  for (auto &param : parameters->arguments)
590  {
591  // Parameters are referenced from the arguments array
592  c.setSymbol(param->name, &args[i++]);
593  }
594  i = 0;
595  for (auto &bound : boundVars)
596  {
597  // Bound variables are stored within the closure object
598  c.setSymbol(bound, &self->boundVars[i++]);
599  }
600  // Interpret the body
601  body->interpret(c);
602  // Return the return value. Make sure it's set back to nullptr after we've
603  // copied it so that we always return null from any method that doesn't
604  // explicitly return.
605  Obj retVal = c.retVal;
606  c.retVal = nullptr;
607  c.isReturning = false;
608  // Pop the symbols off the symbol table (very important, as they reference
609  // our stack frame!)
610  c.popSymbols();
611  return retVal;
612 }

659 Obj BinOp::evaluateExpr(Interpreter::Context &c)
660 {
661  Obj LHS = lhs->evaluate(c);
662  Obj RHS = rhs->evaluate(c);
663  // If this is a comparison, then we're doing a pointer-compare even if
664  // they're objects. If both sides are small integers, then ask the subclass
665  // to look up their integer values.
666  if (isComparison() || (isInteger(LHS) && isInteger(RHS)))
667  {
668  // Ask the subclass to evaluate the expression with integer arguments.
669  // Note that we can't use getInteger() here because the assert that this
670  // really is a small integer is not valid - if we're doing a comparison
671  // then the arguments might be pointers.
672  return createSmallInteger(evaluateWithIntegers(
673  ((intptr_t)LHS) >> 3,
674  ((intptr_t)RHS) >> 3));
675  }
676  Selector sel = lookupSelector(methodName());
677  CompiledMethod mth = compiledMethodForSelector(LHS, sel);
678  return ((Obj(*)(Obj,Selector,Obj))mth)(LHS, sel, RHS);

689 void ClassDecl::interpret(Interpreter::Context &c)
690 {
691  // Construct the new class. The class table persists over the lifetime of
692  // the program, so memory allocated here is never freed.
693  Class *cls = new Class();
694  // Due to the way automatic AST construction works, we'll end up with the
695  // class name in the superclass name field if we don't have a superclass.
696  std::string &clsName = name ? name->name : superclassName->name;
697  cls->superclass = name ? lookupClass(superclassName->name) : nullptr;
698  cls->className = strdup(clsName.c_str());
699  cls->methodCount = methods.size();
700  cls->indexedIVarCount = ivars.size();
701  // Construct the method list, with one Method structure for the metadata for
702  // each method.
703  cls->methodList = new Method[cls->methodCount];
704  Method *method = cls->methodList;
705  for (auto &m : methods)
706  {
707  method->selector = lookupSelector(m->name->name);
708  method->args = m->parameters->arguments.size();
709  // Currently, we only have trampolines for up to 10 arguments. We could
710  // reuse some of the JIT code to generate new ones at run time if
711  // this were intended for production use, but this is okay as an
712  // example.
713  assert(method->args <= 10);
714  // Insert a trampoline for the method
715  method->function = methodTrampolines[method->args];
716  // We retain ownership of the AST node, but the method will contain a
717  // pointer to it.
718  method->AST = m.get();
