141  }

146  {

148  }

153  };

166  return iterator(this, size());
167  }
173  inline char32_t operator[](Index n)
174  {
175  // If the local buffer can satisfy the request, fetch the value
176  if ((n >= buffer_start) && (n < buffer_end))
177  {
178  return buffer[n - buffer_start];
179  }
180  return slowCharacterLookup(n);
181  }
187  Input() : buffer(0), buffer_start(1), buffer_end(0) {}
188  private:
