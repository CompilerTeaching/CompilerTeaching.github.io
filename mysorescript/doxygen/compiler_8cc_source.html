197 CompiledMethod ClosureDecl::compileMethod(Class *cls,
198  Interpreter::SymbolTable &globalSymbols)
199 {
200  auto &params = parameters->arguments.objects();
201  Compiler::Context c(globalSymbols);
202  // Get the type of the method as an LLVM type
203  FunctionType *ClosureInvokeTy = c.getMethodType(cls->indexedIVarCount,
204  params.size());
205  // Create the LLVM function
206  c.F = Function::Create(ClosureInvokeTy, GlobalValue::ExternalLinkage,
207  "invoke", &c.M);
208  // Insert the first basic block and store all of the parameters.
209  BasicBlock *entry = BasicBlock::Create(c.C, "entry", c.F);
210  c.B.SetInsertPoint(entry);
211  // We'll store the arguments into stack allocations so that they can be
212  // written to.
213  auto AI = c.F->arg_begin();
214  // The first two arguments are the two implicit parameters, self and cmd
215  auto selfPtr = c.B.CreateAlloca(c.ObjPtrTy);
216  auto cmdPtr = c.B.CreateAlloca(c.ObjPtrTy);
217  // Add these two stack locations to the symbol table.
218  c.symbols["self"] = selfPtr;
219  c.symbols["cmd"] = cmdPtr;
220  // We're now going to create stack allocations for all of the parameters and
221  // all of the locals. We do this *before* we initialise any of them because
222  // that makes life easier for the LLVM pass that generates SSA form, which
223  // only guarantees to promote allocas that occur before any other
224  // instructions to SSA registers.
225  SmallVector<Value*, 10> paramAllocas;
226  SmallVector<Value*, 10> localAllocas;
227 
228  // Create a stack allocation for each explicit parameter
229  for (size_t i=0 ; i<params.size() ; i++)
230  {
231  paramAllocas.push_back(c.B.CreateAlloca(c.ObjPtrTy));
232  }
233  // Create a stack allocation for each local variable
234  for (size_t i=0 ; i<decls.size() ; i++)
235  {
236  localAllocas.push_back(c.B.CreateAlloca(c.ObjPtrTy));
237  }
238  // Now we'll initialise the parameter allocations
239  auto alloca = paramAllocas.begin();
240  // First store the self pointer in its slot (which is already in the symbol
241  // table)
242  c.B.CreateStore(c.B.CreateBitCast(AI++, c.ObjPtrTy), selfPtr);
243  // The selector is a 32-bit integer, promote it to an object and store it.
244  c.B.CreateStore(getAsObject(c, compileSmallInt(c, c.B.CreateZExt(AI++,
245  c.ObjIntTy))), cmdPtr);
246  for (auto &arg : params)
247  {
248  // Set the name of the stack slot. This isn't necessary, but makes
249  // reading the generated IR a bit easier.
250  (*alloca)->setName(arg->name);
251  // Store the argument in the stack slot.
252  c.B.CreateStore(AI++, *alloca);
253  // Set this slot (specifically, the address of this stack allocation) to
254  // the address used when looking up the name of the argument.
255  c.symbols[arg->name] = *(alloca++);
256  }
257  alloca = localAllocas.begin();
258  // Now do almost the same thing for locals
259  for (auto &local : decls)
260  {
261  (*alloca)->setName(local);
262  // Initialise all local variables with null.
263  c.B.CreateStore(ConstantPointerNull::get(c.ObjPtrTy), *alloca);
264  c.symbols[local] = *(alloca++);
265  }
266  // Next we need to add the instance variables to the symbol table.
267  if (cls->indexedIVarCount)
268  {
269  // This does pointer arithmetic on the first argument to get the address
270  // of the array of instance variables.
271  Value *iVarsArray = c.B.CreateStructGEP(c.F->arg_begin(), 1);
272  for (int i=0 ; i<cls->indexedIVarCount ; i++)
273  {
274  const char *name = cls->indexedIVarNames[i];
275  // Now we do similar arithmetic on the array to get the address of
276  // each instance variable.
277  c.symbols[name] = c.B.CreateStructGEP(iVarsArray, i++, name);
278  }
279  }
280  // Compile the statements in the method.
281  body->compile(c);
282  // If we hit a return statement, it will clear the insert block, so if there
283  // is still a valid insert block then the function ends with an unterminated
284  // basic block.
285  if (c.B.GetInsertBlock() != nullptr)
286  {
287  // Return null if there's no explicit return
288  c.B.CreateRet(ConstantPointerNull::get(c.ObjPtrTy));
289  }
290  // Generate the compiled code.
291  return (CompiledMethod)c.compile();
292 }

295 {
296  auto &params = parameters->arguments.objects();
297  Compiler::Context c(globalSymbols);
298  // Get the LLVM type of the closure invoke function
299  FunctionType *ClosureInvokeTy = c.getClosureType(boundVars.size(),
300  params.size());
301  // Create the LLVM function
302  c.F = Function::Create(ClosureInvokeTy, GlobalValue::ExternalLinkage, "invoke", &c.M);
303  // Insert the first basic block and store all of the parameters.
304  BasicBlock *entry = BasicBlock::Create(c.C, "entry", c.F);
305  c.B.SetInsertPoint(entry);
306  auto AI = c.F->arg_begin();
307  auto selfPtr = c.B.CreateAlloca(c.ObjPtrTy);
308  c.symbols["self"] = selfPtr;
309  SmallVector<Value*, 10> paramAllocas;
310  SmallVector<Value*, 10> localAllocas;
311 
312  for (size_t i=0 ; i<params.size() ; i++)
313  {
314  paramAllocas.push_back(c.B.CreateAlloca(c.ObjPtrTy));
315  }
316  for (size_t i=0 ; i<decls.size() ; i++)
317  {
318  localAllocas.push_back(c.B.CreateAlloca(c.ObjPtrTy));
319  }
320  auto alloca = paramAllocas.begin();
321 
322  c.B.CreateStore(c.B.CreateBitCast(AI++, c.ObjPtrTy), selfPtr);
323  for (auto &arg : params)
324  {
325  (*alloca)->setName(arg->name);
326  c.B.CreateStore(AI++, *alloca);
327  c.symbols[arg->name] = *(alloca++);
328  }
329  alloca = localAllocas.begin();
330  for (auto &local : decls)
331  {
332  (*alloca)->setName(local);
333  c.B.CreateStore(ConstantPointerNull::get(c.ObjPtrTy), *alloca);
334  c.symbols[local] = *(alloca++);
335  }
336  // If this closure refers to external variables then they'll have been
337  // copied into the closure structure. Add those to the symbol table in the
338  // same way that we added instance variables for methods.
339  if (!boundVars.empty())
340  {
341  Value *boundVarsArray = c.B.CreateStructGEP(c.F->arg_begin(), 4);
342  int i=0;
343  for (auto &bound : boundVars)
344  {
345  c.symbols[bound] = c.B.CreateStructGEP(boundVarsArray, i++, bound);
346  }
347  }
348  body->compile(c);
349  if (c.B.GetInsertBlock() != nullptr)
350  {
351  c.B.CreateRet(ConstantPointerNull::get(c.ObjPtrTy));
352  }
353  return c.compile();
354 }

356 Value *ClosureDecl::compileExpression(Compiler::Context &c)
357 {
358  // Make sure that we know what the bound variables are.
359  check();
360  auto &params = parameters->arguments.objects();
361  // Get the type of the invoke function
362  FunctionType *invokeTy = c.getClosureType(boundVars.size(), params.size());
363  // Get the type of the first parameter (a pointer to the closure structure)
364  Type *closurePtrTy = invokeTy->getParamType(0);
365  // The size of the closure is the size of `Closure`, which contains all of
366  // the fields shared by all closures, and then space for all of the bound
367  // variables.
368  size_t closureSize = sizeof(struct Closure) + boundVars.size() * sizeof(Obj);
369  // Insert the `GC_malloc` function (provided by libgc) into the module,
370  // bitcast to return a pointer to our closure type.
371  Constant *allocFn = c.M.getOrInsertFunction("GC_malloc", closurePtrTy,
372  c.ObjIntTy, nullptr);
373  // Allocate GC'd memory for the closure. Note that it would often be more
374  // efficient to do this on the stack, but only if we can either statically
375  // prove that the closure is not captured by anything that is called or if
376  // we can promote it to the heap if it is.
377  Value *closure = c.B.CreateCall(allocFn, ConstantInt::get(c.ObjIntTy,
378  closureSize));
379  // Set the isa pointer to the closure class.
380  c.B.CreateStore(staticAddress(c, &ClosureClass, c.ObjPtrTy),
381  c.B.CreateStructGEP(closure, 0));
382  // Set the parameters pointer to the number of parameters.
383  c.B.CreateStore(compileSmallInt(c, params.size()),
384  c.B.CreateStructGEP(closure, 1));
385  // If we've already compiled the function for this closure, then insert a
386  // pointer to it into the closure, otherwise use the trampoline that calls
387  // back into the interpreter.
388  // Note: This means that if we later compile this closure we should
389  // recompile the enclosing function or the invocation of the closure will be
390  // expensive.
391  ClosureInvoke closureFn = compiledClosure ? compiledClosure :
392  Interpreter::closureTrampolines[params.size()];
393  c.B.CreateStore(staticAddress(c, closureFn, c.ObjPtrTy),
394  c.B.CreateStructGEP(closure, 2));
395  // Set the AST pointer
396  c.B.CreateStore(staticAddress(c, this, c.ObjPtrTy),
397  c.B.CreateStructGEP(closure, 3));
398  // Get a pointer to the array of bound variables
399  Value *boundVarsArray = c.B.CreateStructGEP(closure, 4);
400  int i=0;
401  for (auto &var : boundVars)
402  {
403  // Load each bound variable and then insert it into the closure at the
404  // correct index.
405  c.B.CreateStore(c.B.CreateLoad(c.lookupSymbolAddr(var)),
406  c.B.CreateStructGEP(boundVarsArray, i++, var));
407  }
408  // Add this closure to our symbol table.
409  c.symbols[name->name] = closure;
410  return closure;
411 }
412 Value *Call::compileExpression(Compiler::Context &c)
413 {
414  SmallVector<Value*, 10> args;
415  // Compile the expression that evaluates to the object being called.
416  Value *obj = getAsObject(c, callee->compileExpression(c));
417  assert(obj);
418  // For both closure and method invocations, the object being invoked is the
419  // first argument
420  args.push_back(obj);
421  // If this is a method invocation, then the next argument is the selector.
422  if (method)
423  {
424  Selector sel = lookupSelector(method->name);
425  args.push_back(ConstantInt::get(c.SelTy, sel));
426  }
427  // Now add each of the explicit arguments.
428  auto &argsAST = arguments->arguments;
429  for (auto &arg : argsAST)
430  {
431  args.push_back(getAsObject(c, arg->compileExpression(c)));
432  }
433  // If there's no method, then we're trying to invoke a closure.
434  if (!method)
435  {
436  // Get the closure invoke type.
437  FunctionType *invokeFnTy = c.getClosureType(0, args.size() - 1);
438  // The only field we care about in the closure is the invoke pointer,
439  // so define enough of the structure to get that (two pointers before
440  // it)
441  Type *Fields[3] =
442  { c.ObjPtrTy, c.ObjPtrTy, invokeFnTy->getPointerTo() };
443  // Get the type of a pointer to the closure object
444  Type *closurePtrTy = StructType::create(Fields)->getPointerTo();
445  // Cast the called object to a closure
446  Value *closure = c.B.CreateBitCast(obj, closurePtrTy);
447  // Compute the address of the pointer to the closure invoke function
448  Value *invokeFn = c.B.CreateStructGEP(closure, 2);
449  // Load the address of the invoke function
450  invokeFn = c.B.CreateLoad(invokeFn);
451  // Insert the call
452  return c.B.CreateCall(invokeFn, args, "call_closure");
453  }
454  // If we are invoking a method, then we must first look up the method, then
455  // call it.
456  FunctionType *methodType = c.getMethodType(0, args.size() - 2);
457  // Get the lookup function
458  Constant *lookupFn = c.M.getOrInsertFunction("compiledMethodForSelector",
459  methodType->getPointerTo(), obj->getType(), c.SelTy, nullptr);
460  // Insert the call to the function that performs the lookup. This will
461  // always return *something* that we can call, even if it's just a function
462  // that reports an error.
463  Value *methodFn = c.B.CreateCall2(lookupFn, obj, args[1]);
464  // Call the method
465  return c.B.CreateCall(methodFn, args, "call_method");
466 }

494 void IfStatement::compile(Compiler::Context &c)
495 {
496  // Compute the condition
497  Value *cond = condition->compileExpression(c);
498  // Create the basic block that we'll branch to if the condition is false and
499  // after executing if the condition is true.
500  BasicBlock *cont = BasicBlock::Create(c.C, "if.cont", c.F);
501  // Create the block that contains the body of the if statement
502  BasicBlock *ifBody = BasicBlock::Create(c.C, "if.body", c.F);
503  // Cast the condition to a small int. We aren't unconditionally
504  // interpreting it as a small int, we just want to be able to do some
505  // arithmetic on it...
506  cond = getAsSmallInt(c, cond);
507  // Now right shift it by 3. If it is an object pointer, it will now be 0 if
508  // it were a null pointer before the shift. If it is a small integer, then
509  // it will now contain its numeric value. Either way, a value of 0 means
510  // either a zero integer or null, and a value of anything else means a valid
511  // object or non-zero integer.
512  cond = c.B.CreateLShr(cond, ConstantInt::get(c.ObjIntTy, 3));
513  // Create a comparison with 0 that we can then branch on
514  cond = c.B.CreateIsNotNull(cond);
515  // Branch to the body if it's not 0, to the continuation block if it is
516  c.B.CreateCondBr(cond, ifBody, cont);
517  // Compile the body of the if statement.
518  c.B.SetInsertPoint(ifBody);
519  body->compile(c);
520  // If the if statement didn't end with a return, then return control flow to
521  // the block after the if statement.
522  if (c.B.GetInsertBlock() != nullptr)
523  {
524  c.B.CreateBr(cont);
525  }
526  // Continue from the end
527  c.B.SetInsertPoint(cont);
528 }
529 void WhileLoop::compile(Compiler::Context &c)
530 {
531  // Create three blocks, one for the body of the test (which we will
532  // unconditionally branch back to at the end of the body), one for the loop
533  // body (which we will skip if the condition is false) and one for the end,
534  // which we will branch to after the loop has finished.
535  BasicBlock *condBlock = BasicBlock::Create(c.C, "while.cond", c.F);
536  BasicBlock *cont = BasicBlock::Create(c.C, "while.cont", c.F);
537  BasicBlock *whileBody = BasicBlock::Create(c.C, "while.body", c.F);
538  // Unconditionally branch to the block for the condition and compile it
539  c.B.CreateBr(condBlock);
540  c.B.SetInsertPoint(condBlock);
541  // Compile the condition expression
542  Value *cond = condition->compileExpression(c);
543  // Convert it to an integer and test that it isn't null
544  cond = getAsSmallInt(c, cond);
545  cond = c.B.CreateLShr(cond, ConstantInt::get(c.ObjIntTy, 3));
546  cond = c.B.CreateIsNotNull(cond);
547  // Branch into the loop body or past it, depending on the condition value.
548  c.B.CreateCondBr(cond, whileBody, cont);
549  c.B.SetInsertPoint(whileBody);
550  // Compile the body of the loop.
551  body->compile(c);
552  // Unconditionally branch back to the condition to check it again.
553  c.B.CreateBr(condBlock);
554  // Set the insert point to the block after the loop.
555  c.B.SetInsertPoint(cont);
556 }

667 typedef std::function<Value*(Compiler::Context &c, Value*, Value*)> BinOpFn;
675 Value *compileBinaryOp(Compiler::Context &c, Value *LHS, Value *RHS,
676  BinOpFn intFn, const char *slowCallFnName)
677 {
678  // Get the two operands as integer values
679  Value *LHSInt = getAsSmallInt(c, LHS);
680  Value *RHSInt = getAsSmallInt(c, RHS);
681  // And them together. If they are both small ints, then the low bits of the
682  // result will be 001.
683  Value *isSmallInt = c.B.CreateAnd(LHSInt, RHSInt);
684  // Now mask off the low 3 bits.
685  isSmallInt = c.B.CreateAnd(isSmallInt, ConstantInt::get(c.ObjIntTy, 7));
686  // If the low three bits are 001, then it is a small integer
687  isSmallInt = c.B.CreateICmpEQ(isSmallInt, ConstantInt::get(c.ObjIntTy, 1));
688  // Create three basic blocks, one for the small int case, one for the real
689  // object case, and one for when the two join together again.
690  BasicBlock *cont = BasicBlock::Create(c.C, "cont", c.F);
691  BasicBlock *small = BasicBlock::Create(c.C, "int", c.F);
692  BasicBlock *obj = BasicBlock::Create(c.C, "obj", c.F);
693  // If both arguments are small integers, jump to the small int block,
694  // otherwise fall back to the other case.
695  c.B.CreateCondBr(isSmallInt, small, obj);
696 
697  // Now emit the small int code:
698  c.B.SetInsertPoint(small);
699  // Shift both values right by 3 to give primitive integer values
700  LHSInt = c.B.CreateAShr(LHSInt, ConstantInt::get(c.ObjIntTy, 3));
701  RHSInt = c.B.CreateAShr(RHSInt, ConstantInt::get(c.ObjIntTy, 3));
702  // Invoke the function passed by the caller to insert the correct operation.
703  Value *intResult = intFn(c, LHSInt, RHSInt);
704  // Now cast the result to an object and branch to the continue block
705  intResult = getAsObject(c, compileSmallInt(c, intResult));
706  c.B.CreateBr(cont);
707 
708  // Next we'll handle the real object case.
709  c.B.SetInsertPoint(obj);
710  // Call the function that handles the object case
711  Value *objResult = c.B.CreateCall2(c.M.getOrInsertFunction(slowCallFnName,
712  c.ObjPtrTy, LHS->getType(), RHS->getType(), nullptr), LHS,
713  RHS);
714  // And branch to the continuation block
715  c.B.CreateBr(cont);
716 
717  // Now that we've handled both cases, we need to unify the flow control and
718  // provide a single value
719  c.B.SetInsertPoint(cont);
720  // Construct a PHI node to hold the result.
721  PHINode *result = c.B.CreatePHI(intResult->getType(), 2, "sub");
722  // Set its value to the result of whichever basic block we arrived from
723  result->addIncoming(intResult, small);
724  result->addIncoming(objResult, obj);
725  // Return the result
726  return result;
727 }
