42 static void logTimeSince(clock_t c1, const char *msg)
43 {
44  if (!enableTiming) { return; }
45  clock_t c2 = clock();
46  struct rusage r;
47  getrusage(RUSAGE_SELF, &r);
48  fprintf(stderr, "%s took %f seconds. Peak used %ldKB.\n", msg,
49  ((double)c2 - (double)c1) / (double)CLOCKS_PER_SEC, r.ru_maxrss/1024);
50 }

64 int main(int argc, char **argv)
65 {
66  clock_t c1;
67  // Are we in read-evaluate-print-loop mode?
68  bool repl = false;
69  // Are memory usage statistics requested?
70  bool memstats = false;
71  // What file should we print?
72  const char *file = nullptr;
73  if (argc < 1)
74  {
75  usage(argv[0]);
76  return EXIT_FAILURE;
77  }
78  int c;
79  // Parse the options that we understand
80  while ((c = getopt(argc, argv, "hmitf:")) != -1)
81  {
82  switch (c)
83  {
84  case 'i':
85  repl = true;
86  break;
87  case 'f':
88  file = optarg;
89  break;
90  case 't':
91  enableTiming = true;
92  break;
93  case 'm':
94  memstats = true;
95  break;
96  case 'h':
97  usage(argv[0]);
98  break;
99  }
100  }
101  c1 = clock();
102  //Initialise the garbage collection library. This must be called before
103  //any objects are allocated.
104  GC_init();
105 
106  // Set up a parser and interpreter context to use.
107  Parser::MysoreScriptParser p;
108  Interpreter::Context C;
109  // Log the time taken for all of the program setup.
110  logTimeSince(c1, "Setup");
111  // The AST for the program loaded from a file, if there is one
112  std::unique_ptr<AST::Statements> ast = 0;
113  // If a filename was specified, then try to parse and execute it.
114  if (file)
115  {
116  c1 = clock();
117  // Open the file
118  pegmatite::AsciiFileInput input(open(file, O_RDONLY));
119  pegmatite::ErrorList el;
120  c1 = clock();
121  // Parse one or more statements, report errors if there are any
122  if (!p.parse(input, p.g.statements, p.g.ignored, el, ast))
123  {
124  std::cerr << "errors: \n";
125  for (auto &err : el)
126  {
127  std::cerr << "line " << err.start.line
128  << ", col " << err.finish.col << ": ";
129  std::cerr << "syntax error" << std::endl;
130  }
131  return EXIT_FAILURE;
132  }
133  logTimeSince(c1, "Parsing program");
134  c1 = clock();
135  // Now interpret the parsed
136  ast->interpret(C);
137  logTimeSince(c1, "Executing program");
138  }
139  // Keep all of the ASTs that we've parsed in the REPL environment in case
140  // anything is referencing them.
141  std::vector<std::unique_ptr<AST::Statements>> replASTs;
142  // As long as we're in REPL mode, read, evaluate and loop - we don't
143  // actually print the result, so technically this is REL...
144  while (repl)
145  {
146  c1 = clock();
147  GC_gcollect();
148  logTimeSince(c1, "Garbage collection");
149  std::string buffer;
150  // Print the prompt
151  std::cout << "\nMysoreScript> ";
152  // Get a line
153  std::getline(std::cin, buffer);
154  // If it was an empty line, exit REPL mode
155  if (buffer.size() == 0)
156  {
157  repl = false;
158  break;
159  }
160  // Parse the line
161  pegmatite::StringInput input(buffer);
162  std::unique_ptr<AST::Statements> ast = 0;
163  pegmatite::ErrorList el;
164  c1 = clock();
165  if (!p.parse(input, p.g.statements, p.g.ignored, el, ast))
166  {
167  std::cerr << "errors: \n";
168  for (auto &err : el)
169  {
170  std::cerr << "line " << err.start.line
171  << ", col " << err.finish.col << ": ";
172  std::cerr << "syntax error" << std::endl;
173  }
174  continue;
175  }
176  logTimeSince(c1, "Parsing program");
177  c1 = clock();
178  // Interpret the resulting AST
179  ast->interpret(C);
180  logTimeSince(c1, "Executing program");
181  // Keep the AST around - it may contain things that we refer to later
182  // (e.g. functions / classes).
183  replASTs.push_back(std::move(ast));
184  }
185  // Print some memory usage stats, if requested.
186  if (memstats)
187  {
188  fprintf(stderr, "Allocated %lld bytes during execution.\n", (long
189  long)GC_get_total_bytes());
190  fprintf(stderr, "GC heap size: %lld bytes.\n",
191  (long long)GC_get_heap_size());
192  ast = nullptr;
193  replASTs.clear();
194  GC_gcollect_and_unmap();
195  fprintf(stderr, "After collection, GC heap size: %lld bytes.\n",
196  (long long)GC_get_heap_size());
197  }
198  return 0;
199 }
